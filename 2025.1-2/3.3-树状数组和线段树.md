### 3.3-树状数组和线段树

## 代码部分

### 树状数组

```cpp
#include <iostream>
#include <vector>

using namespace std;

class FenwickTree {
private:
    vector<int> tree;

    int lowbit(int x) {
        return x & -x;
    }

public:
    FenwickTree(int n) {
        tree.resize(n + 1, 0);
    }

    void update(int idx, int val) {
        while (idx < tree.size()) {
            tree[idx] += val;
            idx += lowbit(idx);
        }
    }

    int query(int idx) {
        int sum = 0;
        while (idx > 0) {
            sum += tree[idx];
            idx -= lowbit(idx);
        }
        return sum;
    }

    int queryRange(int l, int r) {
        return query(r) - query(l - 1);
    }
};

int main() {
    vector<int> arr = {1, 3, 5, 7, 9, 11};
    int n = arr.size();

    FenwickTree fenwickTree(n);

    for (int i = 0; i < n; ++i) {
        fenwickTree.update(i + 1, arr[i]);
    }
    cout << "Sum of elements in range [2, 4]: " << fenwickTree.queryRange(2, 4) << endl;

    fenwickTree.update(3, 2);
    cout << "Sum of elements in range [2, 4] after update: " << fenwickTree.queryRange(2, 4) << endl;

    return 0;
}
```



### 线段树

```cpp
#include <iostream>
#include <vector>

using namespace std;

class SegmentTree {
private:
    vector<int> tree;
    int n;
    void buildTree(const vector<int>& arr, int node, int start, int end) {
        if (start == end) {
            tree[node] = arr[start];
        } else {
            int mid = start + end >> 2;
            int leftChild = node << 1 | 1;
            int rightChild = node << 1 | 2;
            buildTree(arr, leftChild, start, mid);
            buildTree(arr, rightChild, mid + 1, end);
            tree[node] = tree[leftChild] + tree[rightChild];
        }
    }

    int query(int node, int start, int end, int l, int r) {
        if (r < start || l > end) {
            return 0;
        }
        if (l <= start && r >= end) {
            return tree[node];
        }
        int mid = (start + end) / 2;
        int leftChild = node << 1 | 1;
        int rightChild = node << 1 | 2;
        int leftSum = query(leftChild, start, mid, l, r);
        int rightSum = query(rightChild, mid + 1, end, l, r);
        return leftSum + rightSum;
    }

    void update(int node, int start, int end, int idx, int val) {
        if (start == end) {
            tree[node] = val;
        } else {
            int mid = (start + end) / 2;
            int leftChild = node << 1 | 1;
            int rightChild = node << 1 | 2;
            if (idx <= mid) {
                update(leftChild, start, mid, idx, val);
            } else {
                update(rightChild, mid + 1, end, idx, val);
            }
            tree[node] = tree[leftChild] + tree[rightChild];
        }
    }

public:
    // 构造函数。
    SegmentTree(const vector<int>& arr) {
        n = arr.size();
        tree.resize(4 * n);
        buildTree(arr, 0, 0, n - 1);
    }

    int queryRange(int l, int r) {
        return query(0, 0, n - 1, l, r);
    }

    void updateValue(int idx, int val) {
        update(0, 0, n - 1, idx, val);
    }
};

int main() {
    vector<int> arr = {1, 3, 5, 7, 9, 11};
    
    SegmentTree st(arr);
    cout << "Sum of values in range [1, 3]: " << st.queryRange(1, 3) << endl;

    st.updateValue(2, 6);
    cout << "Sum of values in range [1, 3] after update: " << st.queryRange(1, 3) << endl;

    return 0;
}
```



## 练习

[1264. 动态求连续区间和](https://www.acwing.com/problem/content/1266/)

树状数组（运行时间：235 ms）：

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 100010;

int a[N], tr[N], n, m;

int lowbit(int x)
{
    return x & -x;
}

void add(int x, int y)
{
    for (int i = x; i <= n; i += lowbit(i)) tr[i] += y;
}


int query(int x)
{
    int res = 0;
    for (int i = x; i > 0; i -= lowbit(i)) res += tr[i];
    return res;
}

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    for (int i = 1; i <= n; i++) add(i, a[i]);

    while (m--)
    {
        int op, x, y;
        scanf("%d%d%d", &op, &x, &y);
        int ret;
        if (!op)
        {
            if (x == 1)
                ret = query(y);
            else
                ret = query(y) - query(x - 1);
            printf("%d\n", ret);
        }
        else
        {
            add(x, y);
        }
    }

    return 0;
}
```

线段树（运行时间：830 ms）：

```cpp
#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 100010;

int w[N], n, m;
struct Node
{
    int l, r;
    int sum;
}tri[N * 4];

void pushup(int u)
{
    tri[u].sum = tri[u << 1].sum + tri[u << 1 | 1].sum;
    return;
}

void build(int u, int l, int r)
{
    if (l == r) tri[u] = {l, r, w[r]};
    else
    {
        tri[u] = {l, r};
        int mid = l + r >> 1;
        build(u << 1, l, mid), build(u << 1 | 1, mid + 1, r);
        pushup(u);
    }
}

int query(int u, int l, int r)
{
    if (tri[u].l >= l && tri[u].r <= r) return tri[u].sum;
    else
    {
        int mid = tri[u].l + tri[u].r >> 1;
        int sum = 0;
        if (l <= mid) sum = query(u << 1, l, r);
        if (r > mid) sum += query(u << 1 | 1, l, r);
        return sum;
    }
}

void modify(int u, int x, int v)
{
    if (tri[u].l == tri[u].r) tri[u].sum += v;
    else
    {
        int mid = tri[u].l + tri[u].r >> 1;
        if (x > mid) modify(u << 1 | 1, x, v);
        else modify(u << 1, x, v);
        pushup(u);
    }
}

int main()
{
    scanf("%d%d", &n, &m);
    for (int i = 1; i <= n; i++) scanf("%d", &w[i]);
    build(1, 1, n);
    while (m--)
    {
        int k, a, b;
        scanf("%d%d%d", &k, &a, &b);
        if (!k) cout <<  query(1, a, b) << endl;
        else modify(1, a, b);
    }
    return 0;
}
```
